{{- /* Create an icon. @param {string} name The name of the icon. @param {string} attributes The attributes of the icon. or @param {string} 0 The name of the icon. @example {{< icon name="github" >}} @example {{< icon "github" >}} */ -}} {{- $name := .Get "name" | default (.Get 0) -}} {{- $attributes := .Get "attributes" | default "height=1em" }} {{- $icon := partial "utils/icon.html" (dict "name" $name "attributes" $attributes) -}} {{- $icon | safeHTML -}}
