{{- /* Gallery (PhotoSwipe v5) */ -}} {{- $galleryBase := "" -}} {{- $useDefaultCdn := true -}} {{- $hasGalleryBase := false -}} {{- with site.Params.gallery.base -}} {{- $galleryBase = . -}} {{- $useDefaultCdn = false -}} {{- $hasGalleryBase = true -}} {{- end -}} {{- $galleryCoreJsAsset := "" -}} {{- with site.Params.gallery.js -}} {{- $galleryCoreJsAsset = . -}} {{- end -}} {{- $galleryLightboxJsAsset := "" -}} {{- with site.Params.gallery.lightboxJs -}} {{- $galleryLightboxJsAsset = . -}} {{- end -}} {{- $galleryCssAsset := "" -}} {{- with site.Params.gallery.css -}} {{- $galleryCssAsset = . -}} {{- end -}} {{- /* If JS is set without base, use local asset loading for the configured modules. */ -}} {{- if and $useDefaultCdn (or (ne $galleryCoreJsAsset "") (ne $galleryLightboxJsAsset "")) -}} {{- $useDefaultCdn = false -}} {{- end -}} {{- if $useDefaultCdn -}} {{- $galleryBase = "https://cdn.jsdelivr.net/npm/photoswipe@5/dist" -}} {{- end -}} {{- $isRemoteBase := or (strings.HasPrefix $galleryBase "http://") (strings.HasPrefix $galleryBase "https://") -}} {{- $minSuffix := cond hugo.IsProduction ".min" "" -}} {{- $galleryCssLink := "" -}} {{- if and $galleryCssAsset (or (not $hasGalleryBase) (not $isRemoteBase)) -}} {{- with resources.Get $galleryCssAsset -}} {{- $css := . | fingerprint -}} {{- $galleryCssLink = printf "" $css.RelPermalink $css.Data.Integrity -}} {{- else -}} {{- errorf "PhotoSwipe css asset not found at %q" $galleryCssAsset -}} {{- end -}} {{- else if $isRemoteBase -}} {{- $cssPath := cond (ne $galleryCssAsset "") $galleryCssAsset "photoswipe.css" -}} {{- $galleryCssUrl := urls.JoinPath $galleryBase $cssPath -}} {{- with try (resources.GetRemote $galleryCssUrl) -}} {{- with .Err -}} {{- errorf "Could not retrieve PhotoSwipe css file from %s. Reason: %s." $galleryCssUrl . -}} {{- else with .Value -}} {{- with resources.Copy "css/photoswipe.css" . -}} {{- $css := . | fingerprint -}} {{- $galleryCssLink = printf "" $css.RelPermalink $css.Data.Integrity -}} {{- end -}} {{- end -}} {{- end -}} {{- else if $galleryCssAsset -}} {{- with resources.Get $galleryCssAsset -}} {{- $css := . | fingerprint -}} {{- $galleryCssLink = printf "" $css.RelPermalink $css.Data.Integrity -}} {{- else -}} {{- errorf "PhotoSwipe css asset not found at %q" $galleryCssAsset -}} {{- end -}} {{- end -}} {{- $galleryCssLink | safeHTML -}} {{- /* JS: PhotoSwipe v5 is ESM-only. We need two files: lightbox and core. */ -}} {{- $lightboxUrl := "" -}} {{- $coreUrl := "" -}} {{- if $isRemoteBase -}} {{- $lightboxPath := cond (ne $galleryLightboxJsAsset "") $galleryLightboxJsAsset (printf "photoswipe-lightbox.esm%s.js" $minSuffix) -}} {{- $corePath := cond (ne $galleryCoreJsAsset "") $galleryCoreJsAsset (printf "photoswipe.esm%s.js" $minSuffix) -}} {{- $lightboxRemote := urls.JoinPath $galleryBase $lightboxPath -}} {{- with try (resources.GetRemote $lightboxRemote) -}} {{- with .Err -}} {{- errorf "Could not retrieve PhotoSwipe lightbox js from %s. Reason: %s." $lightboxRemote . -}} {{- else with .Value -}} {{- with resources.Copy (printf "js/photoswipe-lightbox.esm%s.js" $minSuffix) . -}} {{- $lightboxUrl = (. | fingerprint).RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- $coreRemote := urls.JoinPath $galleryBase $corePath -}} {{- with try (resources.GetRemote $coreRemote) -}} {{- with .Err -}} {{- errorf "Could not retrieve PhotoSwipe core js from %s. Reason: %s." $coreRemote . -}} {{- else with .Value -}} {{- with resources.Copy (printf "js/photoswipe.esm%s.js" $minSuffix) . -}} {{- $coreUrl = (. | fingerprint).RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- else -}} {{- if $galleryLightboxJsAsset -}} {{- with resources.Get $galleryLightboxJsAsset -}} {{- $lightboxUrl = (. | fingerprint).RelPermalink -}} {{- else -}} {{- errorf "PhotoSwipe lightbox js asset not found at %q" $galleryLightboxJsAsset -}} {{- end -}} {{- end -}} {{- if $galleryCoreJsAsset -}} {{- with resources.Get $galleryCoreJsAsset -}} {{- $coreUrl = (. | fingerprint).RelPermalink -}} {{- else -}} {{- errorf "PhotoSwipe core js asset not found at %q" $galleryCoreJsAsset -}} {{- end -}} {{- end -}} {{- end -}} {{- if or (not $lightboxUrl) (not $coreUrl) -}} {{- errorf "PhotoSwipe JS configuration is incomplete. Set both params.gallery.js and params.gallery.lightboxJs, set params.gallery.base to an absolute URL, or omit JS overrides to use the default CDN." -}} {{- end -}} {{- if and $lightboxUrl $coreUrl -}} {{- end -}}
